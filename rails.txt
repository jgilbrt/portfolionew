document.addEventListener('DOMContentLoaded', function() {
    // Add staggered animation delays to tools
    const tools = document.querySelectorAll('.tool-item');
    tools.forEach((tool, index) => {
        tool.style.animationDelay = `${index * 0.1}s`;
        tool.classList.add('animate__animated', 'animate__fadeInUp');
    });

    // Add hover effect enhancement
    const cards = document.querySelectorAll('.education-card, .tool-item');
    cards.forEach(card => {
        card.addEventListener('mouseenter', function() {
            this.style.transform = this.classList.contains('education-card') ? 'translateY(-8px)' : 'translateY(-4px)';
        });

        card.addEventListener('mouseleave', function() {
            this.style.transform = 'translateY(0)';
        });
    });

    // Smooth scroll for anchor links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
            e.preventDefault();
            const target = document.querySelector(this.getAttribute('href'));
            if (target) {
                target.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            }
        });
    });
});
</script>

  <script>
    document.addEventListener('DOMContentLoaded', function() {
      const filterButtons = document.querySelectorAll('#filter-btn-group .filter-btn');
      const projectItems = document.querySelectorAll('#projectsGrid .project-item');
      const projectsGrid = document.getElementById('projectsGrid');

      filterButtons.forEach(btn => {
        btn.addEventListener('click', function () {
          filterButtons.forEach(b => b.classList.remove('active'));
          this.classList.add('active');

          const filter = this.getAttribute('data-filter');

          projectsGrid.style.pointerEvents = 'none';

          projectItems.forEach(item => {
            const category = item.getAttribute('data-category');
            const shouldShow = (filter === 'all' || category === filter);

            if (shouldShow) {
              item.classList.remove('hide');
            } else {
              item.classList.add('hide');
            }
          });

          setTimeout(() => {
              projectsGrid.style.pointerEvents = 'auto';
          }, 500);

        });
      });
    });
  </script>

document.addEventListener('DOMContentLoaded', function() {
    // Add staggered animation delays to tools
    const tools = document.querySelectorAll('.tool-item');
    tools.forEach((tool, index) => {
        tool.style.animationDelay = `${index * 0.1}s`;
        tool.classList.add('animate__animated', 'animate__fadeInUp');
    });

    // Add hover effect enhancement
    const cards = document.querySelectorAll('.education-card, .tool-item');
    cards.forEach(card => {
        card.addEventListener('mouseenter', function() {
            this.style.transform = this.classList.contains('education-card') ? 'translateY(-8px)' : 'translateY(-4px)';
        });

        card.addEventListener('mouseleave', function() {
            this.style.transform = 'translateY(0)';
        });
    });

    // Smooth scroll for anchor links
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function (e) {
            e.preventDefault();
            const target = document.querySelector(this.getAttribute('href'));
            if (target) {
                target.scrollIntoView({
                    behavior: 'smooth',
                    block: 'start'
                });
            }
        });
    });
});
</script>
